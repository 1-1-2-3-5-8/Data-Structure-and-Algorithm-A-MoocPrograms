#include <iostream>
#include <string.h>
using namespace std;

#define maxsize 1000

char trans(char ch)
{
	if (ch == 'z')
		return 'a';
	if (ch == 'Z')
		return 'A';
	if (ch >= 'a'&&ch <= 'y')
		return ch + 1;
	if (ch >= 'A'&&ch <= 'Y')
		return ch + 1;
	return ch;
}

int main()
{
	int n;
	scanf("%d", &n);
	char str[maxsize + 1];
	char c;
	scanf("%c", &c);//¹ýÂË»»ÐÐ
	while (n--)
	{
		cin.getline(str, maxsize);
		int len = strlen(str);
		for (int i = 0; i < len; i++)
			printf("%c", trans(str[i]));
		printf("\n");
	}
	return 0;
}
